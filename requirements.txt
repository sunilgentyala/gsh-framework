# Governed Security Hunting (GSH) Framework
# Python Dependencies - v1.5.0
#
# Core dependencies are intentionally minimal.
# The framework ships with stdlib-only fallbacks for all optional packages.
#
# Install all dependencies:
#   pip install -r requirements.txt
#
# Minimal install (stdlib fallbacks only, no LLM SDK):
#   pip install pyyaml

# ---------------------------------------------------------------------------
# Required
# ---------------------------------------------------------------------------

# YAML policy file parsing (sentinel-policy-default.yaml)
pyyaml>=6.0.1

# ---------------------------------------------------------------------------
# Optional - LLM endpoint integration
# ---------------------------------------------------------------------------

# OpenAI-compatible endpoint client (gsh-probe-eval.py)
# Works with OpenAI, Azure OpenAI, Ollama, LM Studio, and other local servers.
# If not installed, gsh-probe-eval.py falls back to stdlib urllib.
openai>=1.30.0

# ---------------------------------------------------------------------------
# Optional - Enhanced analysis and output
# ---------------------------------------------------------------------------

# Richer statistical analysis for behavioral baselining
# If not installed, the framework uses built-in pure-Python implementations.
numpy>=1.26.0

# HTTP request library, used by the Splunk HEC and Elastic bulk output
# adapters (adapters/splunk_hec.py, adapters/elastic_bulk.py). If not
# installed, siem_output: splunk|elastic logs a warning and falls back to
# local file output.

# Windows Event Log output adapter (adapters/windows_eventlog.py). The
# "; sys_platform == 'win32'" marker means pip skips this line entirely on

# If not installed (or not on Windows), siem_output: windows_eventlog logs
# a warning and falls back to local file output.
pywin32>=306; sys_platform == "win32"
